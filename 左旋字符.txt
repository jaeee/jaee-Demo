void left_rotate(char * str,int n)
{
	assert(str != NULL);
	char *ptr = str;
	while (n--)
	{
		str = ptr;
		int sz = strlen(str);
		char temp;
		temp = *str;
		char *left = str+1;
		while (--sz)
		{
			*str++ = *left++;
		}
		*str = temp;
	}
}

{
	char arr[] = "AABCD";
